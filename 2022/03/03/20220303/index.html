剑指 Offer 09. 用两个栈实现队列

很简单的原理，一个负责出一个负责进，注意下细节就行 

class CQueue {
public:
    stack<int> s1;
    stack<int> s2;
    int a;

    CQueue() {}

    void appendTail(int value) {
        s1.push(value);
    }

    int deleteHead() {
        if((s1.empty())&&(s2.empty())){
            return -1;
        }
        if(s2.empty()){
            while(!s1.empty()){
                s2.push(s1.top());
                s1.pop();
            }
        }
        a = s2.top();
        s2.pop();
        return a;
    }
};

剑指 Offer 30. 包含min函数的栈

同简单，一个存栈另外一个辅助栈负责min（）

class MinStack {
public:
    /** initialize your data structure here. */

    MinStack() {}

    stack<int> s1;
    stack<int> s2;

    void push(int x) {
        s1.push(x);
        if(s2.empty()||s2.top()>=x){
            s2.push(x);
        }
    }

    void pop() {
        if(s2.top()==s1.top()){
            s2.pop();
        }
        s1.pop();
    }

    int top() {
        return s1.top();
    }

    int min() {
        return s2.top();
    }
};

剑指 Offer 24. 反转链表

刚提到翻转链表… 用了个递归的思路 递归可以理解为F(n)=F(n-1)+{F(n-2）+…}
递归思想感觉不能想着完全摊开,要用子问题关系+每次相同执行操作+结束条件(底层)去做.
还要注意递归和操作关系，哪些是先操作后递归，哪些是先递归后操作的，和最底层是如何执行.

class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head == nullptr || head->next == nullptr) {  
            return head;
        }
        ListNode* node = reverseList(head->next);
        head->next->next = head;
        head->next = nullptr;
        return node;  
    }
};

剑指 Offer 35. 复杂链表的复制

主要是熟悉unordered_map的简单机理
这个是2次处理的方法，还有个拼接法没看. 

/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* next;
    Node* random;

    Node(int _val) {
        val = _val;
        next = NULL;
        random = NULL;
    }
};
*/
class Solution {
public:
    Node* copyRandomList(Node* head) {
        //如果头节点为空就直接return
        if(head == nullptr) return nullptr; 
        //从头开始遍历
        Node* cur = head;
        //底层为Hashmap
        unordered_map<Node*, Node*> map;

        //复制各节点，并建立 “原节点 -> 新节点” 的 Map 映射
        while(cur != nullptr) {
            map[cur] = new Node(cur->val);
            cur = cur->next;
        }

        //造完各节点后再回到头节点开始处理next和randow
        cur = head;

        //构建新链表的 next 和 random 指向
        while(cur != nullptr) {
            map[cur]->next = map[cur->next];
            map[cur]->random = map[cur->random];
            cur = cur->next;
        }
        //返回新链表的头节点
        return map[head];
    }
};

class Solution {
public:
    string replaceSpace(string s) {
        int count = 0,len=s.size();
        for (int i=0;i<s.size();i++) {
            if (s[i]==32) count++;
        }
        s.resize(s.size() + 2 * count);
        for(int i = len - 1, j = s.size() - 1; i < j; i--, j--) {
            if (s[i] != 32)
                s[j] = s[i];
            else {
                s[j - 2] = '%';
                s[j - 1] = '2';
                s[j] = '0';
                j -= 2;
            }
        }
        return s;
    }
};

剑指 Offer 53 - I. 在排序数组中查找数字 I

二分 

class Solution {
public:
    int Search(vector<int>& nums, int target, bool lower) {
        int left = 0, right = (int)nums.size() - 1, ans = (int)nums.size();
        while (left <= right) {
            int mid = (left + right) / 2;
            if (nums[mid] > target || (lower && nums[mid] >= target)) {
                right = mid - 1;
                ans = mid;
            } else {
                left = mid + 1;
            }
        }
        return ans;
    }

    int search(vector<int>& nums, int target) {
        int leftIdx = Search(nums, target, true);
        int rightIdx = Search(nums, target, false) - 1;
        if (leftIdx <= rightIdx && rightIdx < nums.size() && nums[leftIdx] == target && nums[rightIdx] == target) {
            return rightIdx - leftIdx + 1;
        }
        return 0;
    }
};

剑指 Offer 21. 调整数组顺序使奇数位于偶数前面

可以按着快排模板改,一样的二分思想

快排模板： 

void quick_sort(int q[], int l, int r)
{
    if (l >= r) return;

    int i = l - 1, j = r + 1, x = q[l + r >> 1];
    while (i < j)
    {
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j), quick_sort(q, j + 1, r);
}

本题代码：

class Solution {
public:
    vector<int> exchange(vector<int>& nums)
    {
        int i = 0, j = nums.size() - 1;
        while (i < j)
        {
            while(i < j && (nums[i] & 1) == 1) i++;
            while(i < j && (nums[j] & 1) == 0) j--;
            swap(nums[i], nums[j]);
        }
        return nums;
    }
};

剑指 Offer 58 - I. 翻转单词顺序

字符串语句通解
https://leetcode-cn.com/problems/fan-zhuan-dan-ci-shun-xu-lcof/solution/yi-ge-mo-ban-shua-bian-suo-you-zi-fu-chu-x6vh/

class Solution {
public:
    string reverseWords(string s){
        if (s.empty())
            return "";

        //<模板> 
        //这一步将每个空格隔开的子字符串重新合并入res中
        s += " "; 
        string temp = "";  //临时字符串
        vector<string> res; //存放字符串的数组
        for (char ch : s)  //遍历字符句子
        {
            if (ch == ' ') //遇到空格
            {
                if (!temp.empty()) //临时字符串非空
                {
                    res.push_back(temp);
                    temp.clear();  //清空临时字符串
                }
            }
            else
                temp += ch; 
        }


        s.clear();
        reverse(res.begin(), res.end());
        for (string &str : res)
            s += str + ' ';
        s.pop_back();
        return s;
    }
};

剑指 Offer 25. 合并两个排序的链表

递归法 

class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        if(l1 == nullptr) return l2;
        if(l2 == nullptr) return l1;
        if(l1->val <= l2->val) {
            l1->next = mergeTwoLists(l1->next, l2);
            return l1;
        } else {
            l2->next = mergeTwoLists(l1, l2->next);
            return l2;
        }
    }
};

伪头节点迭代法

class Solution {
public: 
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode* dum = new ListNode(0); 
        ListNode* cur = dum;
        while(l1 != nullptr && l2 != nullptr) {
            if(l1->val < l2->val) {
                cur->next = l1;
                l1 = l1->next;
            }
            else {
                cur->next = l2;
                l2 = l2->next;
            }
            cur = cur->next;
        }
        cur->next = l1 != nullptr ? l1 : l2;
        return dum->next;
    }
};

剑指 Offer 52. 两个链表的第一个公共节点

class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        ListNode *node1 = headA;
        ListNode *node2 = headB;

        while (node1 != node2) {
            node1 = node1 != NULL ? node1->next : headB;
            node2 = node2 != NULL ? node2->next : headA;
        }

    }
};

剑指 Offer 32 - I. 从上到下打印二叉树

层序遍历用队列即可 每次对一个节点处理后 将起左右子节点插入队列中（若存在） 

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> levelOrder(TreeNode* root) {
        vector<int> res;
        queue<TreeNode*> que;
        if(root!=nullptr) que.push(root);
        while(!que.empty()){
            TreeNode* node = que.front();
            que.pop();
            res.push_back(node->val);
            if(node->left!=nullptr) que.push(node->left);
            if(node->right!=nullptr) que.push(node->right);
        }
        return res;
    }
};

class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        queue<TreeNode*> que;
        if(root!=nullptr) que.push(root);
        while(!que.empty()){
            vector<int> temp;
            for(int i=que.size();i>0;i--){
                TreeNode* node = que.front();
                que.pop();
                temp.push_back(node->val);
                if(node->left!=NULL) que.push(node->left);
                if(node->right!=NULL) que.push(node->right);
            }
            res.push_back(temp);
        }
        return res;
    }
};

剑指 Offer 26. 树的子结构

递归各子节点进行对比 

class Solution {
public:
    bool isSubStructure(TreeNode* A, TreeNode* B) {
        if(B == NULL || A == NULL)
            return false;
        //遍历A中每个节点，A树中任一节点包含B就能返回true
        return iscontain(A, B) || isSubStructure(A->left, B) || isSubStructure(A->right, B);
    }
    //包含：以A为根的数是否包含B（必须从A开始）
    bool iscontain(TreeNode* A, TreeNode* B){
        if(B == NULL)
            return true;
        if(A == NULL || A->val != B->val)
            return false;
        return iscontain(A->left, B->left) && iscontain(A->right, B->right);
    }
};

        if(L == nullptr && R == nullptr) return true;

剑指 Offer 42. 连续子数组的最大和

<状态定义>: dp[i]代表以元素nums[i]为结尾的连续子数组最大和
<转移方程>: dp[i]=dp[i-1]+max(nums[i],0)
<初始状态>: dp[0]=nums[0]，即以nums[0]结尾的连续子数组最大和为 nums[0]
<返回值>: dp数组中最大的值 用res更新即可

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
    int res = nums[0];
    for(int i=0;i<nums.size();i++){
        if(i!=0) nums[i]+=max(nums[i-1],0);
        res=max(res,nums[i]);
    }
     return res;
    }
};

剑指 Offer 47. 礼物的最大价值

<状态定义>: dp[i][j]代表以grid[i][j]为结尾的礼物价值最大和
<转移方程>: dp[i][j]=grid[i][j]+max(dp[i-1][j],dp[i][j-1])
<初始状态>: dp[0][0]=grid[0][0]
<返回值>: 右下角 dp[grid.size()-1][grid[0].size()-1]
其实可以通过直接在grid上面改达到节省空间从O(MN)到O(1)
也可以在新建数组加多一行一列 直接避免边界判断 

class Solution {
public:
    int maxValue(vector<vector<int>>& grid) {
        int dp[grid.size()][grid[0].size()];
        for(int i=0;i<grid.size();i++){
            for(int j=0;j<grid[0].size();j++){
                if(i==0 && j==0) dp[i][j] = grid[i][j];
                else if(i==0 && j!=0) dp[0][j] = dp[0][j-1] + grid[0][j];
                else if(j==0 && i!=0) dp[i][0] = dp[i-1][0] + grid[i][0];
                else if(i!=0 && j!=0){
                    dp[i][j] = max(dp[i-1][j],dp[i][j-1]) + grid[i][j];
                }
            }
        }
        return dp[grid.size()-1][grid[0].size()-1];
    }
};

剑指 Offer 46. 把数字翻译成字符串

理解题意花了好一段时间 其实就只有1~25能被翻译 判断区间即可 

class Solution {
public:
    int translateNum(int num) {
        string str = to_string(num);
        int len = str.size();
        if(len < 2) return len;
        vector<int> dp(len+1);
        dp[1] = 1;
        dp[0] = 1;
        for(int i = 2;i <= len;i++){
            if(str[i-2] == '1' 
                || (str[i-2] == '2' && str[i-1] <= '5')) dp[i] = dp[i-2]+dp[i-1];
            else dp[i] = dp[i-1];
        }
        return dp[len];
    }
};

剑指 Offer 48. 最长不含重复字符的子字符串

这题dp和滑动窗口都可以 有点难
关键其实就是判断左端点的重复字符是否在当前的最长字符串中 

class Solution {
public:
    // 双指针/滑动窗口，
    int lengthOfLongestSubstring(string s) {
        unordered_map<char,int> dic;
        int i = -1, res = 0;
        for(int j = 0; j < s.size(); j++){
            if(dic.count(s[j])) i = max(i,dic.find(s[j])->second);
            dic[s[j]]=j;
            res = max(res,j - i);
        }
        return res;
    }
};
